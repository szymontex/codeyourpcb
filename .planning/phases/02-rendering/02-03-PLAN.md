2. Create `viewer/build-wasm.sh`:
```bash
#!/bin/bash
set -e

# Build WASM module with wasm-pack
# Output to viewer/pkg for Vite to find

cd "$(dirname "$0")/.."

wasm-pack build crates/cypcb-render \
  --target web \
  --out-dir ../viewer/pkg \
  --out-name cypcb_render

echo "WASM build complete: viewer/pkg/"
```

Make executable: `chmod +x viewer/build-wasm.sh`

3. Add npm script to package.json:
```json
{
  "scripts": {
    "build:wasm": "./build-wasm.sh",
    "dev": "vite",
    "build": "npm run build:wasm && vite build"
  }
}
```

4. Run the build: `cd viewer && npm run build:wasm`

The `--target web` flag generates ES modules suitable for browser import.
Output goes to viewer/pkg/ which Vite can serve.

```typescript
import type { BoardSnapshot } from './types';

// PcbEngine wrapper type (matches wasm-bindgen output)
export interface PcbEngine {
  load_source(source: string): string;
  get_snapshot(): BoardSnapshot;
  query_point(x_nm: number, y_nm: number): string[];
  free(): void;
}

let wasmModule: any = null;
let engineInstance: PcbEngine | null = null;

export async function loadWasm(): Promise<PcbEngine> {
  if (engineInstance) {
    return engineInstance;
  }

  // Dynamic import of wasm-pack generated module
  const wasm = await import('../pkg/cypcb_render.js');

  // Initialize WASM (required for web target)
  await wasm.default();

  wasmModule = wasm;

  // Create engine instance
  engineInstance = new wasm.PcbEngine() as PcbEngine;

  return engineInstance;
}

export function getEngine(): PcbEngine | null {
  return engineInstance;
}

// Helper to load source and get snapshot in one call
export function loadAndSnapshot(source: string): { snapshot: BoardSnapshot; errors: string } | null {
  if (!engineInstance) return null;

  const errors = engineInstance.load_source(source);
  const snapshot = engineInstance.get_snapshot();

  return { snapshot, errors };
}
```

Key points:
- Use dynamic import() for WASM module (works with Vite)
- Call wasm.default() to initialize (required for --target web)
- Cache engine instance (avoid recreating, prevents memory leaks)
- Export helpers for common operations

```typescript
import { loadWasm, loadAndSnapshot } from './wasm';
import type { BoardSnapshot } from './types';

// Test source for verification
const TEST_SOURCE = `
version 1
board test {
  size 50mm x 30mm
  layers 2
}
component R1 resistor "0402" {
  value "10k"
  at 10mm, 15mm
}
component C1 capacitor "0402" {
  value "100nF"
  at 20mm, 15mm
}
`;

async function init() {
  const status = document.getElementById('status')!;
  const canvas = document.getElementById('pcb-canvas') as HTMLCanvasElement;
  const container = document.getElementById('canvas-container')!;

  // Resize canvas to container
  function resize() {
    canvas.width = container.clientWidth;
    canvas.height = container.clientHeight;
  }
  resize();
  window.addEventListener('resize', resize);

  status.textContent = 'Loading WASM...';

  try {
    const engine = await loadWasm();
    status.textContent = 'WASM loaded, parsing test source...';

    const result = loadAndSnapshot(TEST_SOURCE);
    if (!result) {
      status.textContent = 'Failed to load source';
      return;
    }

    if (result.errors) {
      console.warn('Parse/sync errors:', result.errors);
      status.textContent = `Errors: ${result.errors.split('\n')[0]}`;
    } else {
      status.textContent = 'Ready';
    }

    // Log snapshot for verification
    console.log('BoardSnapshot:', result.snapshot);
    console.log('Board:', result.snapshot.board);
    console.log('Components:', result.snapshot.components.length);

    // Verify data structure
    if (result.snapshot.board) {
      console.log(`Board: ${result.snapshot.board.name} ${result.snapshot.board.width_nm}nm x ${result.snapshot.board.height_nm}nm`);
    }
    for (const comp of result.snapshot.components) {
      console.log(`  ${comp.refdes}: ${comp.value} at (${comp.x_nm}, ${comp.y_nm}), ${comp.pads.length} pads`);
    }

  } catch (err) {
    console.error('WASM init failed:', err);
    status.textContent = `Error: ${err}`;
  }
}

init().catch(console.error);
```
