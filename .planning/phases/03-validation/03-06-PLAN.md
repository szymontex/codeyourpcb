   /// Rule that checks all drill holes meet minimum size.
   pub struct MinDrillSizeRule;

   impl DrcRule for MinDrillSizeRule {
       fn name(&self) -> &'static str {
           "min-drill-size"
       }

       fn check(
           &self,
           world: &BoardWorld,
           rules: &DesignRules,
       ) -> Vec<DrcViolation> {
           let mut violations = Vec::new();
           let min_drill = rules.min_drill_size;

           // Query all entities with Pad component that has a drill
           let ecs = world.ecs();

           for (entity, pad, pos) in ecs.query::<(Entity, &Pad, &Position)>().iter(ecs) {
               if let Some(drill) = pad.drill {
                   if drill < min_drill {
                       violations.push(DrcViolation::drill_size(
                           entity,
                           drill,
                           min_drill,
                           pos.point,
                       ));
                   }
               }
           }

           violations
       }
   }
   ```

   /// Rule that checks all component pins have net connections.
   pub struct UnconnectedPinRule;

   impl DrcRule for UnconnectedPinRule {
       fn name(&self) -> &'static str {
           "unconnected-pin"
       }

       fn check(
           &self,
           world: &BoardWorld,
           _rules: &DesignRules,
       ) -> Vec<DrcViolation> {
           let mut violations = Vec::new();
           let lib = FootprintLibrary::new();

           let ecs = world.ecs();

           // Query components that have net connections info
           for (entity, refdes, footprint_ref, nets, pos) in
               ecs.query::<(Entity, &RefDes, &FootprintRef, &NetConnections, &Position)>()
                   .iter(ecs)
           {
               // Get footprint definition
               let Some(footprint) = lib.get(footprint_ref.as_str()) else {
                   continue; // Unknown footprint - skip (already caught by sync)
               };

               // Check each pad has a net connection
               for pad in &footprint.pads {
                   // NOTE: NetConnections.pin_net() method exists in
                   // cypcb_world::components::electrical (line 260)
                   if nets.pin_net(&pad.number).is_none() {
                       violations.push(DrcViolation::unconnected_pin(
                           entity,
                           &pad.number,
                           refdes.as_str(),
                           pos.point,
                       ));
                   }
               }
           }

           violations
       }
   }
   ```
