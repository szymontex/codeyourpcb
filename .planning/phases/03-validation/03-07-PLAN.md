   ```rust
   /// A DRC violation for display in the viewer.
   #[derive(Debug, Clone, Serialize, Deserialize)]
   pub struct ViolationInfo {
       /// Violation type (clearance, drill_size, unconnected_pin, etc.)
       pub kind: String,
       /// Board location X in nanometers
       pub x_nm: i64,
       /// Board location Y in nanometers
       pub y_nm: i64,
       /// Human-readable message
       pub message: String,
   }

   impl ViolationInfo {
       pub fn from_drc(v: &cypcb_drc::DrcViolation) -> Self {
           ViolationInfo {
               kind: format!("{:?}", v.kind).to_lowercase(),
               x_nm: v.location.x.0,
               y_nm: v.location.y.0,
               message: v.message.clone(),
           }
       }
   }
   ```
