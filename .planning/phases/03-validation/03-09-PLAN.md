Create a test file with intentional DRC violations:

```cypcb
version 1

board drc_test {
    size 30mm x 30mm
    layers 2
}

// Two resistors placed too close together (should trigger clearance violation)
component R1 resistor "0402" {
    value "10k"
    at 10mm, 15mm
}

component R2 resistor "0402" {
    value "10k"
    at 10.5mm, 15mm  // Only 0.5mm away - will overlap with 0402 footprint
}

// Component with small drill (if through-hole) - test drill rule
component J1 connector "PIN-HDR-1x2" {
    at 20mm, 15mm
}

// Unconnected component (no net assignments) - test connectivity rule
component C1 capacitor "0805" {
    value "100nF"
    at 15mm, 10mm
    // No net assignments - should trigger unconnected pin
}

// Properly connected component for comparison
component R3 resistor "0805" {
    value "1k"
    at 25mm, 15mm
}

net VCC {
    R3.1
}

net GND {
    R3.2
}
```
