   //! Zone components for keepouts and copper pours.

   use bevy_ecs::prelude::*;
   use cypcb_core::Rect;

   /// Zone type.
   #[derive(Debug, Clone, Copy, PartialEq, Eq, Component)]
   pub enum ZoneKind {
       /// No copper allowed in this region.
       Keepout,
       /// Copper fill zone (pour) - connected to a net.
       CopperPour,
   }

   /// A zone entity (keepout or copper pour).
   #[derive(Debug, Clone, Component)]
   pub struct Zone {
       /// Zone bounds.
       pub bounds: Rect,
       /// Zone type.
       pub kind: ZoneKind,
       /// Layer mask (which layers this zone applies to).
       pub layer_mask: u32,
       /// Optional name for reference.
       pub name: Option<String>,
   }

   impl Zone {
       /// Create a new keepout zone.
       pub fn keepout(bounds: Rect, layer_mask: u32) -> Self {
           Zone {
               bounds,
               kind: ZoneKind::Keepout,
               layer_mask,
               name: None,
           }
       }

       /// Create a new copper pour zone.
       pub fn copper_pour(bounds: Rect, layer_mask: u32) -> Self {
           Zone {
               bounds,
               kind: ZoneKind::CopperPour,
               layer_mask,
               name: None,
           }
       }

       /// Set the zone name.
       pub fn with_name(mut self, name: impl Into<String>) -> Self {
           self.name = Some(name.into());
           self
       }

       /// Check if this is a keepout zone.
       pub fn is_keepout(&self) -> bool {
           self.kind == ZoneKind::Keepout
       }

       /// Check if a point is inside the zone.
       pub fn contains(&self, point: cypcb_core::Point) -> bool {
           self.bounds.contains(point)
       }
   }
   ```

   fn sync_zone(zone_def: &ast::ZoneDef, world: &mut BoardWorld) -> Result<(), SyncError> {
       let bounds = Rect::new(
           Point::new(zone_def.bounds.0.to_nm(), zone_def.bounds.1.to_nm()),
           Point::new(zone_def.bounds.2.to_nm(), zone_def.bounds.3.to_nm()),
       );

       let layer_mask = match zone_def.layer.as_deref() {
           Some("top") => 0b01,
           Some("bottom") => 0b10,
           Some("all") | None => 0b11,
           Some(other) => return Err(SyncError::UnknownLayer(other.to_string())),
       };

       let kind = match zone_def.kind {
           ast::ZoneKind::Keepout => ZoneKind::Keepout,
           ast::ZoneKind::CopperPour => ZoneKind::CopperPour,
       };

       let zone = Zone {
           bounds,
           kind,
           layer_mask,
           name: zone_def.name.as_ref().map(|id| id.0.clone()),
       };

       world.ecs_mut().spawn(zone);

       // Note: For copper pours, we would also need to associate with NetId
       // This is deferred until copper pour filling is implemented

       Ok(())
   }

   //! Keepout zone rule - no copper allowed inside keepout regions.

   use bevy_ecs::prelude::*;
   use cypcb_core::Point;
   use cypcb_world::BoardWorld;
   use cypcb_world::components::{Zone, ZoneKind};
   use crate::presets::DesignRules;
   use crate::violation::DrcViolation;
   use super::DrcRule;

   /// Rule that checks no copper features are inside keepout zones.
   pub struct KeepoutRule;

   impl DrcRule for KeepoutRule {
       fn name(&self) -> &'static str {
           "keepout"
       }

       fn check(
           &self,
           world: &BoardWorld,
           _rules: &DesignRules,
       ) -> Vec<DrcViolation> {
           let mut violations = Vec::new();
           let ecs = world.ecs();
           let spatial = world.spatial();

           // Find all keepout zones
           for (_, zone) in ecs.query::<&Zone>().iter(ecs) {
               if !zone.is_keepout() {
                   continue;
               }

               // Query spatial index for copper features inside zone bounds
               let min = zone.bounds.min();
               let max = zone.bounds.max();

               for entry in spatial.query_region_entries(min, max) {
                   // Check layer overlap
                   if (entry.layer_mask & zone.layer_mask) == 0 {
                       continue;
                   }

                   // Entry is copper on a layer covered by this keepout
                   // Calculate center of entry for location
                   let center = Point::new(
                       cypcb_core::Nm((entry.envelope.lower()[0] + entry.envelope.upper()[0]) / 2),
                       cypcb_core::Nm((entry.envelope.lower()[1] + entry.envelope.upper()[1]) / 2),
                   );

                   violations.push(DrcViolation::keepout(
                       entry.entity,
                       zone.name.as_deref(),
                       center,
                   ));
               }
           }

           violations
       }
   }

   #[cfg(test)]
   mod tests {
       use super::*;

       #[test]
       fn test_no_violation_outside_keepout() {
           // Copper feature outside keepout zone
           // Should produce no violation
       }

       #[test]
       fn test_violation_inside_keepout() {
           // Copper feature inside keepout zone
           // Should produce violation
       }

       #[test]
       fn test_no_violation_different_layer() {
           // Copper on top, keepout on bottom only
           // Should produce no violation
       }
   }
   ```
