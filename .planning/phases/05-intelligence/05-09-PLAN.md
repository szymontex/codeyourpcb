   ```rust
   use cypcb_router::{export_dsn, import_ses, FreeRoutingRunner, RoutingConfig};

   #[derive(Args)]
   pub struct RouteArgs {
       /// Input .cypcb file
       input: PathBuf,

       /// Output .routes file (default: input.routes)
       #[arg(short, long)]
       output: Option<PathBuf>,

       /// Path to freerouting.jar
       #[arg(long, env = "FREEROUTING_JAR")]
       freerouting: Option<PathBuf>,

       /// Timeout in seconds (default: 300)
       #[arg(long, default_value = "300")]
       timeout: u64,
   }
   ```
